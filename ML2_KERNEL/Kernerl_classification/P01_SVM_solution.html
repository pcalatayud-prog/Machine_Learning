def configure_data(X, y, symbol='circle', name='data'):
    # Marker symbols: https://plot.ly/python/reference/#scatter-marker
    
    # retrieve indices of zeroes and ones
    idx0 = y==0
    idx1 = y==1
    
    # define data set 1 plot options
    trace0 = go.Scatter3d(x=X[idx0,0], y=X[idx0,1], z=X[idx0,2],
        mode='markers',
        marker=dict(size=6, symbol=symbol, color='blue', opacity=0.6),
        name="%s 0"%name
    )

    # define data set 2 plot options
    trace1 = go.Scatter3d(x=X[idx1,0], y=X[idx1,1], z=X[idx1,2],
        mode='markers',
        marker=dict(size=6, symbol=symbol, color='red', opacity=0.6),
        name="%s 1"%name
    )
    
    return [trace0,trace1]

def plot_3D(data):

    # set aspect ratio
    scene = dict(aspectmode="manual", aspectratio=dict(x=1, y=1, z=1))

    # define figure properties
    layout = go.Layout(
        scene=scene,
        height=600,
        width=900
    )

    # produce the plot
    fig = go.Figure(data=data,layout=layout)
    iplot(fig)
    
# find misclassified data
idx_correct = y_test == y_pred

[trace0,trace1] = configure_data(X_train,y_train,name='Train')
[trace2,trace3] = configure_data(X_test[idx_correct,:],y_test[idx_correct],symbol='square-open',name='Test correct')
[trace4,trace5] = configure_data(X_test[~idx_correct],y_test[~idx_correct],symbol='x',name='Test errors')

plot_3D([trace0,trace1,trace2,trace3,trace4,trace5])

Exercise: Face Recognition¶

In the previous example we have chosen a number of principal components by hand. In order to find suitable parameters for the SVM, we have applied a grid search in which for each set of parameter values we have ran a cross-validation procedure to determine the fitness of the parameters. In this exercise you will extend the gird search to the number of principal components, using the Pipeline class from scikit-learn.

  1. Build a classifier that consists of a pipeline of PCA and the support vector classifier.
  2. Define a grid containing the values to consider for each of the parametres:
    • 3 values for PCA's n_components
    • 3 values for SVC's C
    • 3 values for SVC's gamma
  3. Use GridSearchCV to find the best set of values for these 3 parameters.
  4. Apply the classifier with the best parameters to the test set and plot the classification_report and the confusion_matrix.

# 1. Pipeline

from sklearn.pipeline import make_pipeline
from sklearn.decomposition import PCA

pca = PCA(whiten=True, svd_solver='randomized', random_state=0)
svc = SVC(kernel='rbf', class_weight='balanced')
pipe = make_pipeline(pca, svc)
